        (init) edge node {} (empty)

                {$+:+$} (mb)

                {\textipa{t}$:$\textipa{t}} (t)
        (s) edge [loop left] node {\textipa{s}$:$\textipa{s}} ()

            edge [loop above] node {\textipa{s}$:$\textipa{S}} ()

            edge [bend left=5] node [below right]

            edge node {} (s)
        (t) edge [loop right] node {} ()
            edge [bend left=5] node [below right]
                {\textipa{s}$:$\textipa{s}} (s)
            edge [bend left=5] node [below right]
                {\textipa{s}$:$\textipa{s}} (mb)
            edge [loop right] node {} ()
